# javascript-if-else-digit
JavaScript If and else digit 
<body>
		<p id="demo"></p>
	<script>
		//digit spelling
		//0-zero, one-1...9-nine, 10 not a valid digit
		var digit= prompt("Enter any digit :");
		if(digit==0){
			document.write("zero");
		}
		else if(digit==1){
			document.write("One");
		}
		else if(digit==2){
			document.write("Two");
		}
		else if(digit==3){
			document.write("Three");
		}
		else if(digit==4){
			document.write("Four");
		}
		else if(digit==5){
			document.write("five");
		}
		else if(digit==6){
			document.write("Six");
		}
		else if(digit==7){
			document.write("Seven");
		}
		else if(digit==8){
			document.write("Eight");
		}
		else if(digit==9){
			document.write("Nine");
		}else{
			document.write("Not a digit");
		}
	</script>
	</body>
